// 1. Solving problems using array functions on rest countries data.

var req = new XMLHttpRequest();
req.responseType = "json";
req.open(
  "GET",
  "https://raw.githubusercontent.com/rvsp/restcountries-json-data/master/res-countries.json",
  true
);
req.onload = function () {
  var jsonResponse = req.response;
  // console.log(jsonResponse);

  // a. Get all the countries from Asia continent /region using Filter function
  console.log(jsonResponse.filter((country) => country.region === "Asia"));

  //   b. Get all the countries with a population of less than 2 lakhs using Filter function
  console.log(jsonResponse.filter((country) => country.population < 200000));

  //   c. Print the following details name, capital, flag using forEach function

  jsonResponse.forEach((country) =>
    console.log(country.name + " , " + country.capital + " , " + country.flag)
  );

  //   d. Print the total population of countries using reduce function
  console.log(
    jsonResponse.reduce(
      (count, country) => count + parseInt(country.population),
      0
    )
  );

  // e. Print the country which uses US Dollars as currency
  console.log(
    jsonResponse.filter((country) => country.currencies[0].code === "USD")
  );
};
req.send(null);
